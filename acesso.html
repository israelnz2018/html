function showPrimeiroAcesso() { document.getElementById("loginSection").style.display = "none"; document.getElementById("primeiroAcesso").style.display = "block"; } function showLogin() { document.getElementById("primeiroAcesso").style.display = "none"; document.getElementById("loginSection").style.display = "block"; } async function trocarSenha() { const email = document.getElementById("email").value.trim(); const senha_provisoria = document.getElementById("senha_provisoria").value.trim(); const nova_senha = document.getElementById("nova_senha").value.trim(); const confirmar_senha = document.getElementById("confirmar_senha").value.trim(); if (nova_senha !== confirmar_senha) { alert("As senhas não coincidem."); return; } try { const resposta = await fetch("https://primary-production-1d53.up.railway.app/webhook/senhaprovisoria", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ email, senha_provisoria, nova_senha }) }); const resultado = await resposta.json(); if (resultado.sucesso || (resultado.mensagem && resultado.mensagem.includes("correta"))) { alert("Senha alterada com sucesso! Faça login."); showLogin(); } else { alert(resultado.mensagem || "Erro na troca de senha."); } } catch (error) { console.error(error); alert("Erro ao processar: " + error.message); } } async function fazerLogin() { const email = document.getElementById('email-login').value.trim(); const senha = document.getElementById('senha-login').value.trim(); const msg = document.getElementById('mensagem-login'); const spinner = document.getElementById('spinner-login'); const botao = document.getElementById('botao-login'); if (!email || !senha) { msg.style.color = "red"; msg.textContent = "❌ Preencha todos os campos."; return; } msg.style.color = "#444"; msg.textContent = "🔄 Verificando..."; spinner.style.display = "block"; botao.disabled = true; botao.style.opacity = "0.6"; try { const userCredential = await firebase.auth().signInWithEmailAndPassword(email, senha); const user = userCredential.user; msg.style.color = "green"; msg.textContent = `✅ Bem-vindo, ${user.email}!`; document.getElementById("login-box").style.display = "none"; document.getElementById("app").style.display = "block"; document.getElementById("app-wrapper").classList.remove("opacity-50", "pointer-events-none"); let displayName = user.displayName || user.email.split("@")[0]; const nomes = displayName.split(" "); const iniciais = nomes[0].charAt(0).toUpperCase() + (nomes[1] ? nomes[1].charAt(0).toUpperCase() : ""); document.getElementById("usuario-info").innerHTML = `
Olá, ${displayName}
`; localStorage.setItem('sessaoAtiva', 'true'); localStorage.setItem('usuarioEmail', user.email); localStorage.setItem('usuarioNome', displayName); } catch (error) { console.error("Erro no login:", error); msg.style.color = "red"; msg.textContent = "❌ Erro ao fazer login: " + error.message; } finally { setTimeout(() => { spinner.style.display = "none"; botao.disabled = false; botao.style.opacity = "1"; }, 1500); } } function recuperarSenha() { const email = prompt("Digite seu e-mail para recuperar a senha:"); if (!email) return; firebase.auth().sendPasswordResetEmail(email) .then(() => { alert("📧 Se este e-mail estiver cadastrado, você receberá um link para redefinir sua senha."); }) .catch((error) => { console.error("Erro ao enviar email de redefinição:", error); alert("❌ Erro ao enviar email de redefinição: " + error.message); }); } function deslogarAluno() { firebase.auth().signOut().then(() => { localStorage.removeItem('sessaoAtiva'); localStorage.removeItem('usuarioEmail'); localStorage.removeItem('usuarioNome'); document.getElementById("app").style.display = "none"; document.getElementById("login-box").style.display = "block"; document.getElementById("mensagem-login").textContent = ""; }).catch((error) => console.error("Erro ao deslogar:", error)); }
